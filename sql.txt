create table administrator(Admin_ID number constraint admin_id_pk primary key, fname varchar2(100) constraint fname_not_null not null, lname varchar2(100),address1 varchar2(200) constraint address1_not_null not null,address2 varchar2(200) constraint address2_not_null not null, d_o_r varchar2(20),d_o_b varchar2(20),gender char constraint gender_check check(gender='m'or gender='f'), mail_id varchar2(100) constraint mail_id_uk unique, mob_no number(10),user_name varchar2(100) constraint username_uk unique not null, password varchar2(100) constraint password_not_null not null);
